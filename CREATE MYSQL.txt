create database walmart;

use walmart;

CREATE TABLE GRAFOT (
  GRAFO_ID INT NOT NULL AUTO_INCREMENT,
  nomeMapa VARCHAR(100) NOT NULL,
  primary key(GRAFO_ID)
);

CREATE TABLE VERTICET (
  VERTICE_ID INT NOT NULL AUTO_INCREMENT,
  GRAFO_ID INT NOT NULL,
  NOME_VERTICE VARCHAR(100) NOT NULL,
  primary key(VERTICE_ID),
  FOREIGN KEY (GRAFO_ID) 
        REFERENCES GRAFOT(GRAFO_ID)
        ON DELETE CASCADE
);


CREATE TABLE ARESTAT (
  ARESTA_ID INT NOT NULL AUTO_INCREMENT,
  VERTICE_ID INT NOT NULL,
  DESTINO VARCHAR(100) NOT NULL,
  DISTANCIA DOUBLE(8,2) NOT NULL,
  primary key(ARESTA_ID),
  FOREIGN KEY (VERTICE_ID) 
        REFERENCES VERTICET(VERTICE_ID)
        ON DELETE CASCADE
);
